--Write an SQL statement to find the total purchase amount of all orders.

SELECT SUM(PURCHASE_AMOUNT) AS TOTAL_purchase_Amount
FROM ORDERS

--Write an SQL statement to find the average purchase amount of all orders.

SELECT AVG(PURCHASE_AMOUNT) AS TOTAL_purchase_Amount
FROM ORDERS
--Write an SQL statement to get the maximum purchase amount of all the orders.

SELECT MAX(PURCHASE_AMOUNT) AS TOTAL_purchase_Amount
FROM ORDERS

--Write an SQL statement to get the minimum purchase amount of all the orders.

SELECT MIN(PURCHASE_AMOUNT) AS TOTAL_purchase_Amount
FROM ORDERS
